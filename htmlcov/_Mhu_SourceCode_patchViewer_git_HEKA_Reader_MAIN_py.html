17class Struct(object): 

18 """High-level wrapper around struct.Struct that makes it a bit easier to  

19 unpack large, nested structures.  

20 """ 

21 field_info = None 

22 size_check = None 

23 _fields_parsed = None 

24 

25 def __init__(self, data, endian='<'): 

26 """Read the structure from *data* and return an ordered dictionary of  

27 fields. 

28  

29 *data* may be a string or file. 

30 *endian* may be '<' or '>' 

31 """ 

32 field_info = self._field_info() 

33 if not isinstance(data, (str, bytes)): 

34 data = data.read(self._le_struct.size) 

35 if endian == '<': 

36 items = self._le_struct.unpack(data) 

37 elif endian == '>': 

38 items = self._be_struct.unpack(data) 

39 else: 

40 raise ValueError('Invalid endian: %s' % endian) 

41 

42 fields = collections.OrderedDict() 

43 

44 i = 0 

45 for name, fmt, func in field_info: 

46 # pull item(s) out of the list based on format string 

47 if len(fmt) == 1 or fmt[-1] == 's': 

48 item = items[i] 

49 i += 1 

50 else: 

51 n = int(fmt[:-1]) 

52 item = items[i:i+n] 

53 i += n 

54 

55 # try unpacking sub-structure 

56 if isinstance(func, tuple): 

57 substr, func = func 

58 item = substr(item, endian) 

59 

60 # None here means the field should be omitted 

61 if func is None: 

62 continue 

63 # handle custom massaging function 

64 if func is not True: 

65 item = func(item) 

66 fields[name] = item 

67 setattr(self, name, item) 

68 

69 self.fields = fields 

70 

71 @classmethod 

72 def _field_info(cls): 

73 if cls._fields_parsed is not None: 

74 return cls._fields_parsed 

75 

76 fmt = '' 

77 fields = [] 

78 for items in cls.field_info: 

79 if len(items) == 3: 

80 name, ifmt, func = items 

81 else: 

82 name, ifmt = items 

83 func = True 

84 

85 if isinstance(ifmt, type) and issubclass(ifmt, Struct): 

86 func = (ifmt, func) # instructs to unpack with sub-struct before calling function 

87 ifmt = '%ds' % ifmt.size() 

88 elif len(ifmt) > 1 and re.match(r'\d*[xcbB?hHiIlLqQfdspP]', ifmt) is None: 

89 raise TypeError('Unsupported format string "%s"' % ifmt) 

90 

91 fields.append((name, ifmt, func)) 

92 fmt += ifmt 

93 cls._le_struct = struct.Struct('<' + fmt) 

94 cls._be_struct = struct.Struct('>' + fmt) 

95 cls._fields_parsed = fields 

96 if cls.size_check is not None: 

97# print(cls._le_struct.size, cls.size_check) 

98 assert cls._le_struct.size == cls.size_check 

99 return fields 

100 

101 @classmethod 

102 def size(cls): 

103 cls._field_info() 

104 return cls._le_struct.size 

105 

106 @classmethod 

107 def array(cls, x): 

108 """Return a new StructArray class of length *x* and using this struct  

109 as the array item type. 

110 """ 

111 return type(cls.__name__+'[%d]'%x, (StructArray,), 

112 {'item_struct': cls, 'array_size': x}) 

113 

114 def __repr__(self, indent=0): 

115 indent_str = ' '*indent 

116 r = indent_str + '%s(\n'%self.__class__.__name__ 

117 if not hasattr(self, 'fields'): 

118 r = r[:-1] + '<initializing>)' 

119 return r 

120 for k,v in self.fields.items(): 

121 if isinstance(v, Struct): 

122 r += indent_str + ' %s = %s\n' % (k, v.__repr__(indent=indent+1).lstrip()) 

123 else: 

124 r += indent_str + ' %s = %r\n' % (k, v) 

125 r += indent_str + ')' 

126 return r

128 def get_fields(self): 

129 """Recursively convert struct fields+values to nested dictionaries. 

130 """ 

131 fields = self.fields.copy() 

132 for k,v in fields.items(): 

133 if isinstance(v, StructArray): 

134 fields[k] = [x.get_fields() for x in v.array] 

135 elif isinstance(v, Struct): 

136 fields[k] = v.get_fields() 

137 return fields

140class StructArray(Struct): 

141 item_struct = None 

142 array_size = None 

143 

144 def __init__(self, data, endian='<'): 

145 if not isinstance(data, (str, bytes)): 

146 data = data.read(self.size()) 

147 items = [] 

148 isize = self.item_struct.size() 

149 for i in range(self.array_size): 

150 d = data[:isize] 

151 data = data[isize:] 

152 items.append(self.item_struct(d, endian)) 

153 self.array = items

155 def __getitem__(self, i): 

156 return self.array[i]

200class TreeNode(Struct): 

201 """Struct that also represents a node in a Pulse file tree. 

202 """ 

203 def __init__(self, fh, pul, level=0): 

204 self.level = level 

205 self.children = [] 

206 endian = pul.endian 

207 

208 # The record structure in the file may differ from our expected structure 

209 # due to version differences, so we read the required number of bytes, and 

210 # then pad or truncate before unpacking the record. This will probably 

211 # result in corrupt data in some situations.. 

212 realsize = pul.level_sizes[level] 

213 data = fh.read(realsize) 

214 structsize = self.size() 

215 

216 diff = structsize - realsize 

217 if diff !=0: 

218# print("TreeNode realsize: %g, struct size: %g, size diff: %g" % (realsize, structsize, diff)) 

219 if diff > 0: 

220 data = data + b'\0'*diff 

221 else: 

222 data = data[:structsize] 

223 

224 # initialize struct data 

225 Struct.__init__(self, data, endian) 

226 

227 # Next read the number of children 

228 nchild = struct.unpack(endian + 'i', fh.read(4))[0] 

229 

230 level += 1 

231 if level >= len(pul.rectypes): 

232 return 

233 child_rectype = pul.rectypes[level] 

234 for i in range(nchild): 

235 self.children.append(child_rectype(fh, pul, level)) 

236 

237 def __getitem__(self, i): 

238 return self.children[i] 

239 

240 def __len__(self): 

241 return len(self.children) 

242 

243 def __iter__(self): 

244 return self.children.__iter__() 

245 

246 def __repr__(self, indent=0): 

247 # Return a string describing this structure 

248 ind = ' '*indent 

249 srep = Struct.__repr__(self, indent)[:-1] # exclude final parenthese 

250 srep += ind + ' children = %d,\n' % len(self) 

251 #srep += ind + 'children = [\n' 

252 #for ch in self: 

253 #srep += ch.__repr__(indent=indent+1) + ',\n' 

254 srep += ind + ')' 

255 return srep

721 c= 0

734 data2Read = TotalBytes - dataRead
